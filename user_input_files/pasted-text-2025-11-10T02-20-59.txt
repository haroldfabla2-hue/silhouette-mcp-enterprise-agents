 * Dashboard para monitoreo y anÃ¡lisis del sistema avanzado de gestiÃ³n de contexto

import React, { useState, useEffect } from 'react';
import { motion } from 'framer-motion';
import { 
  Activity, 
  Database, 
  Search, 
  Zap, 
  TrendingUp, 
  Layers, 
  Brain,
  Clock,
  BarChart3,
  Settings,
  RefreshCw,
  Download,
  Upload
} from 'lucide-react';
import { useAdvancedContext } from '../hooks/useAdvancedContext';

const AdvancedContextDashboard: React.FC = () => {
  const {
    session,
    isLoading,
    error,
    contextStats,
    initializeSession,
    addMessage,
    compressNow,
    searchSemantic
  } = useAdvancedContext();

  const [activeTab, setActiveTab] = useState<'overview' | 'layers' | 'semantic' | 'compression' | 'settings'>('overview');
  const [searchQuery, setSearchQuery] = useState('');
  const [searchResults, setSearchResults] = useState<any[]>([]);
  const [autoRefresh, setAutoRefresh] = useState(true);

  // Auto-refresh stats
  useEffect(() => {
    if (autoRefresh && session) {
      const interval = setInterval(() => {
        // Simulate real-time updates
        console.log('ðŸ”„ Auto-refreshing context stats...');
      }, 5000);
      return () => clearInterval(interval);
    }
  }, [autoRefresh, session]);

  const handleSearch = async () => {
    if (!searchQuery.trim() || !session) return;
    
    try {
      const results = await searchSemantic(searchQuery, 10);
      setSearchResults(results);
    } catch (error) {
      console.error('Search failed:', error);
    }
  };

  const getCompressionColor = (ratio: number) => {
    if (ratio > 0.7) return 'text-green-500';
    if (ratio > 0.4) return 'text-yellow-500';
    return 'text-red-500';
  };

  const getCompressionBg = (ratio: number) => {
    if (ratio > 0.7) return 'bg-green-100 border-green-200';
    if (ratio > 0.4) return 'bg-yellow-100 border-yellow-200';
    return 'bg-red-100 border-red-200';
  };

  return (
    <div className="min-h-screen bg-gradient-to-br from-slate-50 to-slate-100 p-6">
      <div className="max-w-7xl mx-auto">
        {/* Header */}
        <div className="mb-8">
          <div className="flex items-center justify-between mb-4">
            <div>
              <h1 className="text-3xl font-bold text-slate-800 mb-2">
                ðŸ§  Advanced Context Management
              </h1>
              <p className="text-slate-600">
                Sistema escalable de gestiÃ³n de contexto para millones de usuarios
              </p>
            </div>
            <div className="flex items-center gap-3">
              <button
                onClick={() => setAutoRefresh(!autoRefresh)}
                className={`px-3 py-2 rounded-lg flex items-center gap-2 text-sm ${
                  autoRefresh 
                    ? 'bg-blue-100 text-blue-700 border border-blue-200' 
                    : 'bg-slate-100 text-slate-600 border border-slate-200'
                }`}
              >
                <RefreshCw className={`w-4 h-4 ${autoRefresh ? 'animate-spin' : ''}`} />
                Auto-refresh
              </button>
              <button
                onClick={compressNow}
                disabled={isLoading}
                className="px-4 py-2 bg-purple-600 text-white rounded-lg hover:bg-purple-700 disabled:opacity-50 flex items-center gap-2"
              >
                <Zap className="w-4 h-4" />
                Comprimir Ahora
              </button>
            </div>
          </div>

          {/* Session Info */}
          {session && (
            <div className="bg-white rounded-lg border border-slate-200 p-4">
              <div className="flex items-center justify-between">
                <div className="flex items-center gap-4">
                  <div className="flex items-center gap-2">
                    <Database className="w-5 h-5 text-blue-600" />
                    <span className="font-medium">Session: {session.sessionId}</span>
                  </div>
                  <div className="flex items-center gap-2">
                    <Clock className="w-4 h-4 text-slate-500" />
                    <span className="text-sm text-slate-600">
                      Last compression: {new Date(session.lastCompression).toLocaleTimeString()}
                    </span>
                  </div>
                </div>
                <div className="flex items-center gap-2">
                  <div className="w-2 h-2 bg-green-500 rounded-full"></div>
                  <span className="text-sm text-slate-600">Active</span>
                </div>
              </div>
            </div>
          )}
        </div>

        {/* Navigation Tabs */}
        <div className="mb-6">
          <div className="flex space-x-1 bg-white rounded-lg p-1 border border-slate-200">
            {[
              { id: 'overview', label: 'Overview', icon: BarChart3 },
              { id: 'layers', label: 'Context Layers', icon: Layers },
              { id: 'semantic', label: 'Semantic Search', icon: Search },
              { id: 'compression', label: 'Compression', icon: TrendingUp },
              { id: 'settings', label: 'Settings', icon: Settings }
            ].map(tab => (
              <button
                key={tab.id}
                onClick={() => setActiveTab(tab.id as any)}
                className={`flex items-center gap-2 px-4 py-2 rounded-md text-sm font-medium transition-colors ${
                  activeTab === tab.id
                    ? 'bg-blue-100 text-blue-700'
                    : 'text-slate-600 hover:text-slate-800 hover:bg-slate-50'
                }`}
              >
                <tab.icon className="w-4 h-4" />
                {tab.label}
              </button>
            ))}
          </div>
        </div>

        {/* Error Display */}
        {error && (
          <motion.div
            initial={{ opacity: 0, y: -10 }}
            animate={{ opacity: 1, y: 0 }}
            className="mb-6 bg-red-50 border border-red-200 text-red-700 px-4 py-3 rounded-lg"
          >
            <strong>Error:</strong> {error}
          </motion.div>
        )}

        {/* Overview Tab */}
        {activeTab === 'overview' && (
          <div className="space-y-6">
            {/* Key Metrics */}
            <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-4 gap-6">
              <motion.div
                initial={{ opacity: 0, y: 20 }}
                animate={{ opacity: 1, y: 0 }}
                className="bg-white rounded-lg border border-slate-200 p-6"
              >
                <div className="flex items-center justify-between mb-4">
                  <div className="p-2 bg-blue-100 rounded-lg">
                    <Database className="w-6 h-6 text-blue-600" />
                  </div>
                  <span className="text-2xl font-bold text-slate-800">
                    {contextStats?.totalTokens?.toLocaleString() || 0}
                  </span>
                </div>
                <h3 className="font-semibold text-slate-700 mb-1">Total Tokens</h3>
                <p className="text-sm text-slate-500">Processed in current session</p>
              </motion.div>

              <motion.div
                initial={{ opacity: 0, y: 20 }}
                animate={{ opacity: 1, y: 0 }}
                transition={{ delay: 0.1 }}
                className={`bg-white rounded-lg border p-6 ${getCompressionBg(contextStats?.compressionRatio || 0)}`}
              >
                <div className="flex items-center justify-between mb-4">
                  <div className="p-2 bg-green-100 rounded-lg">
                    <TrendingUp className="w-6 h-6 text-green-600" />
                  </div>
                  <span className={`text-2xl font-bold ${getCompressionColor(contextStats?.compressionRatio || 0)}`}>
                    {((contextStats?.compressionRatio || 0) * 100).toFixed(1)}%
                  </span>
                </div>
                <h3 className="font-semibold text-slate-700 mb-1">Compression Ratio</h3>
                <p className="text-sm text-slate-500">Efficiency of context storage</p>
              </motion.div>

              <motion.div
                initial={{ opacity: 0, y: 20 }}
                animate={{ opacity: 1, y: 0 }}
                transition={{ delay: 0.2 }}
                className="bg-white rounded-lg border border-slate-200 p-6"
              >
                <div className="flex items-center justify-between mb-4">
                  <div className="p-2 bg-purple-100 rounded-lg">
                    <Layers className="w-6 h-6 text-purple-600" />
                  </div>
                  <span className="text-2xl font-bold text-slate-800">
                    {contextStats?.activeLayers || 0}
                  </span>
                </div>
                <h3 className="font-semibold text-slate-700 mb-1">Active Layers</h3>
                <p className="text-sm text-slate-500">Currently in context window</p>
              </motion.div>

              <motion.div
                initial={{ opacity: 0, y: 20 }}
                animate={{ opacity: 1, y: 0 }}
                transition={{ delay: 0.3 }}
                className="bg-white rounded-lg border border-slate-200 p-6"
              >
                <div className="flex items-center justify-between mb-4">
                  <div className="p-2 bg-orange-100 rounded-lg">
                    <Brain className="w-6 h-6 text-orange-600" />
                  </div>
                  <span className="text-2xl font-bold text-slate-800">
                    {contextStats?.semanticVectors || 0}
                  </span>
                </div>
                <h3 className="font-semibold text-slate-700 mb-1">Semantic Vectors</h3>
                <p className="text-sm text-slate-500">For intelligent retrieval</p>
              </motion.div>
            </div>

            {/* Performance Chart Placeholder */}
            <div className="bg-white rounded-lg border border-slate-200 p-6">
              <h3 className="text-lg font-semibold text-slate-800 mb-4">Context Performance Over Time</h3>
              <div className="h-64 flex items-center justify-center bg-slate-50 rounded-lg">
                <div className="text-center">
                  <BarChart3 className="w-12 h-12 text-slate-400 mx-auto mb-2" />
                  <p className="text-slate-500">Performance chart would be rendered here</p>
                  <p className="text-sm text-slate-400">Real-time metrics visualization</p>
                </div>
              </div>
            </div>
          </div>
        )}

        {/* Layers Tab */}
        {activeTab === 'layers' && session && (
          <div className="space-y-6">
            <div className="bg-white rounded-lg border border-slate-200 p-6">
              <h3 className="text-lg font-semibold text-slate-800 mb-4">Context Layers</h3>
              <div className="space-y-4">
                {session.layers.map((layer, index) => (
                  <motion.div
                    key={layer.id}
                    initial={{ opacity: 0, x: -20 }}
                    animate={{ opacity: 1, x: 0 }}
                    transition={{ delay: index * 0.1 }}
                    className="border border-slate-200 rounded-lg p-4"
                  >
                    <div className="flex items-center justify-between mb-2">
                      <div className="flex items-center gap-3">
                        <span className={`px-2 py-1 rounded text-xs font-medium ${
                          layer.type === 'raw' ? 'bg-blue-100 text-blue-700' :
                          layer.type === 'compressed' ? 'bg-green-100 text-green-700' :
                          'bg-yellow-100 text-yellow-700'
                        }`}>
                          {layer.type.toUpperCase()}
                        </span>
                        <span className="text-sm text-slate-600">{layer.tokens} tokens</span>
                      </div>
                      <div className="flex items-center gap-2">
                        <span className="text-sm text-slate-500">Importance:</span>
                        <div className="w-16 h-2 bg-slate-200 rounded-full overflow-hidden">
                          <div 
                            className="h-full bg-blue-500 rounded-full"
                            style={{ width: `${layer.importance * 100}%` }}
                          />
                        </div>
                        <span className="text-sm text-slate-600 w-8">
                          {(layer.importance * 100).toFixed(0)}%
                        </span>
                      </div>
                    </div>
                    <p className="text-sm text-slate-700 line-clamp-2">
                      {layer.content.substring(0, 150)}...
                    </p>
                    <div className="mt-2 flex items-center gap-4 text-xs text-slate-500">
                      <span>{new Date(layer.timestamp).toLocaleString()}</span>
                      {layer.metadata.compressionRatio && (
                        <span>Compression: {(layer.metadata.compressionRatio * 100).toFixed(1)}%</span>
                      )}
                    </div>
                  </motion.div>
                ))}
              </div>
            </div>
          </div>
        )}

        {/* Semantic Search Tab */}
        {activeTab === 'semantic' && (
          <div className="space-y-6">
            <div className="bg-white rounded-lg border border-slate-200 p-6">
              <h3 className="text-lg font-semibold text-slate-800 mb-4">Semantic Search</h3>
              <div className="flex gap-3 mb-6">
                <input
                  type="text"
                  value={searchQuery}
                  onChange={(e) => setSearchQuery(e.target.value)}
                  onKeyPress={(e) => e.key === 'Enter' && handleSearch()}
                  placeholder="Search through your context using natural language..."
                  className="flex-1 px-4 py-2 border border-slate-300 rounded-lg focus:ring-2 focus:ring-blue-500 focus:border-transparent"
                />
                <button
                  onClick={handleSearch}
                  disabled={!searchQuery.trim()}
                  className="px-6 py-2 bg-blue-600 text-white rounded-lg hover:bg-blue-700 disabled:opacity-50 flex items-center gap-2"
                >
                  <Search className="w-4 h-4" />
                  Search
                </button>
              </div>

              {searchResults.length > 0 && (
                <div className="space-y-4">
                  <h4 className="font-medium text-slate-700">Search Results ({searchResults.length})</h4>
                  {searchResults.map((result, index) => (
                    <motion.div
                      key={result.id}
                      initial={{ opacity: 0, y: 10 }}
                      animate={{ opacity: 1, y: 0 }}
                      transition={{ delay: index * 0.1 }}
                      className="border border-slate-200 rounded-lg p-4"
                    >
                      <div className="flex items-center justify-between mb-2">
                        <span className="px-2 py-1 bg-purple-100 text-purple-700 rounded text-xs font-medium">
                          {result.type.toUpperCase()}
                        </span>
                        <span className="text-sm text-slate-500">
                          Similarity: {(result.similarity * 100).toFixed(1)}%
                        </span>
                      </div>
                      <p className="text-sm text-slate-700">
                        {result.content.substring(0, 200)}...
                      </p>
                    </motion.div>
                  ))}
                </div>
              )}
            </div>
          </div>
        )}

        {/* Compression Tab */}
        {activeTab === 'compression' && session && (
          <div className="space-y-6">
            <div className="grid grid-cols-1 lg:grid-cols-2 gap-6">
              <div className="bg-white rounded-lg border border-slate-200 p-6">
                <h3 className="text-lg font-semibold text-slate-800 mb-4">Compression Settings</h3>
                <div className="space-y-4">
                  <div>
                    <label className="block text-sm font-medium text-slate-700 mb-2">
                      Max Raw Tokens
                    </label>
                    <input
                      type="number"
                      defaultValue={4000}
                      className="w-full px-3 py-2 border border-slate-300 rounded-lg"
                    />
                  </div>
                  <div>
                    <label className="block text-sm font-medium text-slate-700 mb-2">
                      Compression Ratio Target
                    </label>
                    <input
                      type="number"
                      step="0.1"
                      defaultValue={0.6}
                      className="w-full px-3 py-2 border border-slate-300 rounded-lg"
                    />
                  </div>
                  <div>
                    <label className="block text-sm font-medium text-slate-700 mb-2">
                      Auto-compression Interval (minutes)
                    </label>
                    <input
                      type="number"
                      defaultValue={30}
                      className="w-full px-3 py-2 border border-slate-300 rounded-lg"
                    />
                  </div>
                </div>
              </div>

              <div className="bg-white rounded-lg border border-slate-200 p-6">
                <h3 className="text-lg font-semibold text-slate-800 mb-4">Compression Stats</h3>
                <div className="space-y-4">
                  <div className="flex justify-between">
                    <span className="text-slate-600">Current Ratio:</span>
                    <span className={`font-semibold ${getCompressionColor(contextStats?.compressionRatio || 0)}`}>
                      {((contextStats?.compressionRatio || 0) * 100).toFixed(1)}%
                    </span>
                  </div>
                  <div className="flex justify-between">
                    <span className="text-slate-600">Tokens Saved:</span>
                    <span className="font-semibold text-green-600">
                      ~{((contextStats?.totalTokens || 0) * (contextStats?.compressionRatio || 0)).toFixed(0)}
                    </span>
                  </div>
                  <div className="flex justify-between">
                    <span className="text-slate-600">Last Compression:</span>
                    <span className="font-semibold text-slate-800">
                      {new Date(session.lastCompression).toLocaleTimeString()}
                    </span>
                  </div>
                </div>
              </div>
            </div>
          </div>
        )}

        {/* Settings Tab */}
        {activeTab === 'settings' && (
          <div className="space-y-6">
            <div className="bg-white rounded-lg border border-slate-200 p-6">
              <h3 className="text-lg font-semibold text-slate-800 mb-4">System Settings</h3>
              <div className="space-y-6">
                <div>
                  <label className="flex items-center gap-3">
                    <input type="checkbox" defaultChecked className="rounded" />
                    <span className="text-slate-700">Enable auto-compression</span>
                  </label>
                </div>
                <div>
                  <label className="flex items-center gap-3">
                    <input type="checkbox" defaultChecked className="rounded" />
                    <span className="text-slate-700">Enable semantic search</span>
                  </label>
                </div>
                <div>
                  <label className="flex items-center gap-3">
                    <input type="checkbox" className="rounded" />
                    <span className="text-slate-700">Enable real-time monitoring</span>
                  </label>
                </div>
                
                <div className="pt-4 border-t border-slate-200">
                  <h4 className="font-medium text-slate-800 mb-3">Data Management</h4>
                  <div className="flex gap-3">
                    <button className="px-4 py-2 bg-blue-600 text-white rounded-lg hover:bg-blue-700 flex items-center gap-2">
                      <Download className="w-4 h-4" />
                      Export Context
                    </button>
                    <button className="px-4 py-2 bg-green-600 text-white rounded-lg hover:bg-green-700 flex items-center gap-2">
                      <Upload className="w-4 h-4" />
                      Import Context
                    </button>
                  </div>
                </div>
              </div>
            </div>
          </div>
        )}
      </div>
    </div>
  );
};

export default AdvancedContextDashboard;

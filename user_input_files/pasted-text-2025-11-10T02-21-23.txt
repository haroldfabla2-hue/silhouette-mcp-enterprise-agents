/**
 * ADVANCED FLOATING CHAT WITH CONTEXT MANAGEMENT
 * Integración del sistema avanzado de gestión de contexto con el chat
 */

import React, { useState, useRef, useEffect, useCallback } from 'react';
import { motion, AnimatePresence } from 'framer-motion';
import { 
  MessageCircle, 
  X, 
  Send, 
  Bot, 
  User, 
  Sparkles,
  Loader2,
  ChevronDown,
  ChevronUp,
  Settings,
  Brain,
  Zap,
  BarChart3
} from 'lucide-react';
import { useAdvancedContext } from '../../hooks/useAdvancedContext';
import { ChatMessage, PersonalityState } from '../../types';
import { formatDistanceToNow } from 'date-fns';

// Mock user for demo
const MOCK_USER = {
  id: '1',
  email: 'alberto.farah.b@gmail.com',
  name: 'Alberto Farah',
  role: 'admin' as 'admin' | 'worker' | 'client',
  createdAt: new Date().toISOString(),
  lastActive: new Date().toISOString()
};

const AdvancedFloatingChat: React.FC = () => {
  const {
    session,
    isLoading,
    error,
    contextStats,
    initializeSession,
    addMessage,
    getOptimizedContext,
    searchSemantic,
    compressNow
  } = useAdvancedContext();

  const [message, setMessage] = useState('');
  const [isMinimized, setIsMinimized] = useState(false);
  const [showAdvanced, setShowAdvanced] = useState(false);
  const [chatMessages, setChatMessages] = useState<ChatMessage[]>([]);
  const [isTyping, setIsTyping] = useState(false);
  const inputRef = useRef<HTMLInputElement>(null);

  // Initialize context on mount
  useEffect(() => {
    const initializeChat = async () => {
      await initializeSession(MOCK_USER.id);
    };
    initializeChat();
  }, [initializeSession]);

  // Auto-focus input when chat is opened
  useEffect(() => {
    if (!isMinimized && inputRef.current) {
      inputRef.current.focus();
    }
  }, [isMinimized]);

  const handleSendMessage = useCallback(async () => {
    if (!message.trim() || !session) return;

    const userMessage: ChatMessage = {
      id: `msg_${Date.now()}`,
      role: 'user',
      content: message.trim(),
      timestamp: new Date().toISOString()
    };

    // Add to local chat state
    setChatMessages(prev => [...prev, userMessage]);
    setMessage('');
    setIsTyping(true);

    try {
      // Add to advanced context management
      await addMessage(userMessage, 0.8);

      // Get optimized context for response generation
      const optimizedContext = getOptimizedContext(6000);
      
      // Simulate AI response
      setTimeout(() => {
        const aiResponse: ChatMessage = {
          id: `ai_${Date.now()}`,
          role: 'assistant',
          content: generateContextualResponse(userMessage.content, optimizedContext?.content || ''),
          timestamp: new Date().toISOString()
        };
        
        setChatMessages(prev => [...prev, aiResponse]);
        setIsTyping(false);
        
        // Add AI response to context with high importance
        addMessage(aiResponse, 0.9);
      }, 1500);

    } catch (error) {
      console.error('Failed to send message:', error);
      setIsTyping(false);
    }
  }, [message, session, addMessage, getOptimizedContext]);

  const generateContextualResponse = (userMessage: string, context: string): string => {
    // Enhanced response generation using context
    const responses = [
      `Entiendo tu consulta sobre "${userMessage.substring(0, 30)}...". Basándome en nuestro contexto previo, puedo ayudarte de manera más efectiva.`,
      `Interesante punto. Con la información que tenemos en el contexto de nuestra conversación, mi respuesta sería optimizada para tu situación específica.`,
      `Perfecto, veo que esto se relaciona con temas que hemos discutido anteriormente. La gestión avanzada de contexto me permite darte respuestas más personalizadas.`
    ];
    
    return responses[Math.floor(Math.random() * responses.length)];
  };

  const handleKeyPress = (e: React.KeyboardEvent) => {
    if (e.key === 'Enter' && !e.shiftKey) {
      e.preventDefault();
      handleSendMessage();
    }
  };

  const toggleAdvanced = () => {
    setShowAdvanced(!showAdvanced);
  };

  return (
    <div className="fixed bottom-6 right-6 z-50">
      {/* Main Chat Window */}
      <AnimatePresence>
        {!isMinimized && (
          <motion.div
            initial={{ opacity: 0, scale: 0.8, y: 20 }}
            animate={{ opacity: 1, scale: 1, y: 0 }}
            exit={{ opacity: 0, scale: 0.8, y: 20 }}
            className="bg-white rounded-lg shadow-2xl border border-slate-200 mb-4 w-96 h-[600px] flex flex-col"
          >
            {/* Header */}
            <div className="flex items-center justify-between p-4 border-b border-slate-200 bg-gradient-to-r from-blue-50 to-purple-50">
              <div className="flex items-center gap-3">
                <div className="p-2 bg-blue-100 rounded-lg">
                  <Brain className="w-5 h-5 text-blue-600" />
                </div>
                <div>
                  <h3 className="font-semibold text-slate-800">Silhouette AI</h3>
                  <p className="text-xs text-slate-600">Advanced Context Management</p>
                </div>
              </div>
              <div className="flex items-center gap-2">
                <button
                  onClick={toggleAdvanced}
                  className={`p-2 rounded-lg transition-colors ${
                    showAdvanced 
                      ? 'bg-blue-100 text-blue-600' 
                      : 'hover:bg-slate-100 text-slate-600'
                  }`}
                  title="Advanced Features"
                >
                  <Settings className="w-4 h-4" />
                </button>
                <button
                  onClick={() => setIsMinimized(true)}
                  className="p-2 hover:bg-slate-100 rounded-lg text-slate-600"
                >
                  <ChevronDown className="w-4 h-4" />
                </button>
              </div>
            </div>

            {/* Advanced Context Stats */}
            {showAdvanced && contextStats && (
              <motion.div
                initial={{ opacity: 0, height: 0 }}
                animate={{ opacity: 1, height: 'auto' }}
                exit={{ opacity: 0, height: 0 }}
                className="bg-slate-50 border-b border-slate-200 p-3"
              >
                <div className="grid grid-cols-3 gap-3 text-xs">
                  <div className="text-center">
                    <div className="font-semibold text-slate-800">
                      {contextStats.totalTokens.toLocaleString()}
                    </div>
                    <div className="text-slate-500">Tokens</div>
                  </div>
                  <div className="text-center">
                    <div className="font-semibold text-green-600">
                      {(contextStats.compressionRatio * 100).toFixed(0)}%
                    </div>
                    <div className="text-slate-500">Compression</div>
                  </div>
                  <div className="text-center">
                    <div className="font-semibold text-purple-600">
                      {contextStats.semanticVectors}
                    </div>
                    <div className="text-slate-500">Vectors</div>
                  </div>
                </div>
              </motion.div>
            )}

            {/* Messages */}
            <div className="flex-1 overflow-y-auto p-4 space-y-4">
              {chatMessages.length === 0 && (
                <div className="text-center py-8">
                  <Bot className="w-12 h-12 text-slate-400 mx-auto mb-3" />
                  <p className="text-slate-600 mb-2">¡Bienvenido al chat con gestión avanzada de contexto!</p>
                  <p className="text-sm text-slate-500">
                    {contextStats 
                      ? `Contexto: ${contextStats.totalTokens} tokens, ${contextStats.compressionRatio > 0.4 ? 'alta' : 'baja'} compresión`
                      : 'Inicializando sistema de contexto...'
                    }
                  </p>
                </div>
              )}

              {chatMessages.map((msg) => (
                <motion.div
                  key={msg.id}
                  initial={{ opacity: 0, y: 10 }}
                  animate={{ opacity: 1, y: 0 }}
                  className={`flex gap-3 ${msg.role === 'user' ? 'justify-end' : 'justify-start'}`}
                >
                  {msg.role === 'assistant' && (
                    <div className="p-2 bg-blue-100 rounded-lg self-start">
                      <Brain className="w-4 h-4 text-blue-600" />
                    </div>
                  )}
                  <div
                    className={`max-w-[80%] p-3 rounded-lg ${
                      msg.role === 'user'
                        ? 'bg-blue-600 text-white'
                        : 'bg-slate-100 text-slate-800'
                    }`}
                  >
                    <p className="text-sm">{msg.content}</p>
                    <p className="text-xs mt-1 opacity-70">
                      {formatDistanceToNow(new Date(msg.timestamp), { addSuffix: true })}
                    </p>
                  </div>
                  {msg.role === 'user' && (
                    <div className="p-2 bg-blue-600 rounded-lg self-start">
                      <User className="w-4 h-4 text-white" />
                    </div>
                  )}
                </motion.div>
              ))}

              {isTyping && (
                <motion.div
                  initial={{ opacity: 0, y: 10 }}
                  animate={{ opacity: 1, y: 0 }}
                  className="flex gap-3"
                >
                  <div className="p-2 bg-blue-100 rounded-lg">
                    <Brain className="w-4 h-4 text-blue-600" />
                  </div>
                  <div className="bg-slate-100 p-3 rounded-lg">
                    <div className="flex gap-1">
                      <div className="w-2 h-2 bg-slate-400 rounded-full animate-bounce"></div>
                      <div className="w-2 h-2 bg-slate-400 rounded-full animate-bounce" style={{ animationDelay: '0.1s' }}></div>
                      <div className="w-2 h-2 bg-slate-400 rounded-full animate-bounce" style={{ animationDelay: '0.2s' }}></div>
                    </div>
                  </div>
                </motion.div>
              )}
            </div>

            {/* Input */}
            <div className="p-4 border-t border-slate-200">
              <div className="flex gap-2">
                <input
                  ref={inputRef}
                  type="text"
                  value={message}
                  onChange={(e) => setMessage(e.target.value)}
                  onKeyPress={handleKeyPress}
                  placeholder="Escribe tu mensaje..."
                  className="flex-1 px-3 py-2 border border-slate-300 rounded-lg focus:ring-2 focus:ring-blue-500 focus:border-transparent"
                  disabled={isLoading || !session}
                />
                <button
                  onClick={handleSendMessage}
                  disabled={!message.trim() || isLoading || !session}
                  className="p-2 bg-blue-600 text-white rounded-lg hover:bg-blue-700 disabled:opacity-50"
                >
                  <Send className="w-4 h-4" />
                </button>
              </div>
              
              {error && (
                <p className="text-xs text-red-600 mt-2">Error: {error}</p>
              )}
            </div>
          </motion.div>
        )}
      </AnimatePresence>

      {/* Floating Action Button */}
      <motion.button
        onClick={() => setIsMinimized(false)}
        className="bg-gradient-to-r from-blue-600 to-purple-600 text-white p-4 rounded-full shadow-2xl hover:shadow-3xl transition-shadow"
        whileHover={{ scale: 1.05 }}
        whileTap={{ scale: 0.95 }}
      >
        {isMinimized ? (
          <div className="relative">
            <MessageCircle className="w-6 h-6" />
            {contextStats && contextStats.activeLayers > 0 && (
              <div className="absolute -top-2 -right-2 w-4 h-4 bg-green-500 rounded-full flex items-center justify-center">
                <div className="w-2 h-2 bg-white rounded-full"></div>
              </div>
            )}
          </div>
        ) : (
          <X className="w-6 h-6" />
        )}
      </motion.button>

      {/* Context Status Indicator */}
      {session && (
        <div className="absolute -top-2 -left-2 bg-white rounded-full p-2 shadow-lg border border-slate-200">
          <div className="flex items-center gap-2">
            <Zap className="w-3 h-3 text-purple-600" />
            <div className="w-2 h-2 bg-green-500 rounded-full animate-pulse"></div>
          </div>
        </div>
      )}
    </div>
  );
};

export default AdvancedFloatingChat;
